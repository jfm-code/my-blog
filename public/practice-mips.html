3rd type - Jump table in MIPS

  • How to know if the status is defined (1) or not defined (0)
    1. Defined: If a label is declared and associated with a memory address, it is considered "defined." This typically happens when a label is used as a target for a branch instruction, as the starting point of a subroutine, or as the destination for storing data.

      Example:

      loop:   beq $t0, $zero, endLoop

      In this case, loop is defined because it is the target of the branch instruction.

    1. Not Defined: If a label is referenced but not yet associated with an address, it is considered "not defined." This can happen when a label is used before it is declared.

      Example:

      jal subroutine

      If subroutine is not declared before this line, it is "not defined" until its declaration is encountered in the code.

hex2char:  	sw	$t0, saveReg($0)		
            li	$t9, 3			
            jal	hex2char	
saveReg:	  or	$t0, $t1, $0

    • Instruction: hex2char: sw $t0, saveReg($0) → detect hex2char and saveReg

      hex2char0x04001
      saveReg0x04000
    • Instruction: li $t9, 3 → nothing happen
      SymbolValueStatus
      hex2char0x04001
      saveReg0x04000

      SymbolValueStatus
      hex2char0x04001
      saveReg0x04000
      hex2char0x04080
    • Instruction: saveReg: or $t0, $t1, $0 → detect saveReg

      saveReg0x04000
      hex2char0x04080

//pseudo-C
void select_sort(int N)
{
	int i, j, min,max,tmp;
	i=1;
loop1:
	if(i>N/2) goto exit
	min = i;
	max = i;
	j = i+1;
loop2:
	if(j>N-i) goto swap
	if(A[j] <= A[max]) goto condition
	max = j;
	continue;
condition:
	if(A[j] >= A[min]) goto next
	min = j;
next:
	j++;
	goto loop2
swap: 
	tmp = A[i-1];
	A[i-1] = A[min];
	A[min] = tmp;
	tmp = A[N-i];
	A[N-i] = A[max];
	A[max] = tmp;
	i++;
	goto loop1;
exit:
}
#MIPS 
.data
selection_sort:
	A: .word 5, 2, 9, 3, 6, 1, 8, 4, 7, 10
	N: .word 10 #Number of elements in A
.text
	li $t0,1 #Initialize i=1
	la $a0,A #Load the array A into $t1
	li $t2,N #Load the number of elements into $t2
	div $t3, $t2, 2 #Divide N/2
loop1:
	bgt $t0,$t3, exit #If (i > N/2) goto exit
	move $t4, $t0 #min = i
	move $t5, $t0 #max = i
	add $t6, $t0, 1 #j = i+1
loop2:
	sub $t7, $t2, $t0 #Subtract N-i
	bgt $t6, $t7, swap #If (j < N-i) goto swap
	lw $t8, ($a0)+$t6 #Initialize array A with offset $t6 = j
	lw $t9, ($a0)+$t5 #Initialize array A with offset $t5 = 
	ble $t8, $t9, condition #If (A[j] <= A[max]) goto condition
	move $t5, $t6 #max = j
condition:
	lw $s0, ($a0)+$t4 #Initialize array A with offset $t4 = min

void bubble_sort(int N)
{
	int i = N-1;
	while(i>0)
	{
		int pos=0;
		for(int j=0; j<i; i++)
			if(A[j] > A[j+1]
			{
				pos = j;
				int tmp = A[j];
				A[j] = A[j+1];
				A[j+1] = tmp;
			}
		i = pos;
	}
}
//pseudo-C
 void bubble_sort(int N)
{
	int i=N-1;
	while_loop:
		if(i<=0) goto done
		int pos = 0;
		int  j=0;
	for_loop:
		if(j>=i) goto next
		if(A[j] <= A[j+1]) goto next
		pos = j;
		int tmp = A[j];
		A[j] = A[j+1];
		A[j+1] = tmp;
		b for_loop
	next:
		i = pos;
		b while_loop
	done:
}
#MIPS 
.text
.globl bubble_sort

bubble_sort:
    # Function prologue
    sub $sp, $sp, 12  # Allocate space on the stack for i, pos, and j
    sw $ra, 0($sp)   # Save the return address
    sw $a0, 4($sp)   # Save N in the stack
    sw $a1, 8($sp)   # Save A in the stack

    lw $t0, 8($a0)   # Load N from the stack
    subi $t0, $t0, 1 # i = N - 1

while_loop:
    blez $t0, done   # if (i <= 0) goto done

    li $t1, 0        # pos = 0
    li $t2, 0        # j = 0

for_loop:
    bge $t2, $t0, next  # if (j >= i) goto next

    lw $t3, 8($a1)      # Load A[j]
    lw $t4, 12($a1)     # Load A[j + 1]

    ble $t3, $t4, next  # if (A[j] <= A[j + 1]) goto next

    move $t1, $t2       # pos = j
    move $t5, $t3       # tmp = A[j]
    sw $t4, 8($a1)      # A[j] = A[j + 1]
    sw $t5, 12($a1)     # A[j + 1] = tmp

next:
    addi $t2, $t2, 1   # j++
    j for_loop

done:
    lw $ra, 0($sp)      # Restore the return address
    lw $a0, 4($sp)      # Restore N
    lw $a1, 8($sp)      # Restore A
    addi $sp, $sp, 12   # Deallocate stack space
    jr $ra              # Return to the caller

  • Example 7 (exam 1): An integer array X has size called “limit”, write C functions that performs MIPS on the right
//C statement

int func(int limit)
{
	int i,temp;
	temp = X[0];
	for(i=1; i<limit; i++)
	{
		if(X[i] > temp)
			temp = X[i];
	}
	int val = temp;
	return val;
}
#MIPS 
# $t9 has the argument limit
# return value is via $a0 

func: 
	lw $a0, X                # $a0 = temp = X[0]
	li $t0, 1                # $t0 = i = 1
rept:
	bge $t0, $t9, exit       # if (i >= limit) goto exit
	sll $t0, $t0, 2          # moving i inside memory
	lw $a1, X($t0)           # $a1 = X[i]
	bgt $a0, $a1, next       # if(temp > X[i]) goto next
	move $a0, $a1            # else then save $a1 into $a0
next:
	sra $t0, $t0, 2          # moving i inside memory
	addi $t0, $t0, 1         # i++
	b rept
exit:
	jr $ra                   # return temp
